38void FAssetAccessibilityRegistry::OnAssetOpenedInEditor(UObject* OpenedAsset, IAssetEditorInstance* EditorInstance)
39{
40 UE_LOG(LogOpenAccessibility, Log, TEXT("|| AssetRegistry || Asset { %s } Opened In Editor: { %s } ||"), *OpenedAsset->GetName(), *EditorInstance->GetEditorName().ToString());
41
42 // Find Asset Type for correct Parsing.
43 if (UBlueprint* OpenedBlueprint = Cast<UBlueprint>(OpenedAsset))
44 {
45 UE_LOG(LogOpenAccessibility, Log, TEXT("|| AssetRegistry || Asset { %s } Is A Blueprint ||"), *OpenedBlueprint->GetName());
46
47 RegisterBlueprintAsset(OpenedBlueprint);
48 }
49 else if (UMaterial* OpenedMaterial = Cast<UMaterial>(OpenedAsset))
50 {
51 UE_LOG(LogOpenAccessibility, Log, TEXT("|| AssetRegistry || Asset { %s } Is A Material ||"), *OpenedMaterial->GetName());
52
53 RegisterMaterialAsset(OpenedMaterial);
54 }
55 else if (UBehaviorTree* OpenedBehaviorTree = Cast<UBehaviorTree>(OpenedAsset))
56 {
57 UE_LOG(LogOpenAccessibility, Log, TEXT("|| AssetRegistry || Asset { %s } Is A Behavior Tree ||"), *OpenedBehaviorTree->GetName());
58
59 RegisterBehaviorTreeAsset(OpenedBehaviorTree);
60 }
61}

76bool FAssetAccessibilityRegistry::RegisterGraphAsset(const UEdGraph* InGraph)
77{
78 if (!InGraph->IsValidLowLevel())
79 return false;
80
81 GraphAssetIndex.Add(InGraph->GraphGuid, MakeShared<FGraphIndexer>(InGraph));
82
83 for (auto& ChildGraph : InGraph->SubGraphs)
84 {
85 if (!RegisterGraphAsset(ChildGraph))
86 {
87 UE_LOG(LogOpenAccessibility, Error, TEXT("|| AssetRegistry || Error When Logging Child Graph: { %s } From Parent: { %s }||"), *ChildGraph->GetName(), *InGraph->GetName())
88
89 return false;
90 }
91 }
92
93 return true;
94}

189void FAssetAccessibilityRegistry::RegisterBlueprintAsset(const UBlueprint* InBlueprint)
190{
191 // Register the Blueprint's Graphs
192 TArray<UEdGraph*> Graphs;
193
194 InBlueprint->GetAllGraphs(Graphs);
195 for (auto& Graph : Graphs)
196 {
197 RegisterGraphAsset(Graph);
198 }
199
200 // Register the Blueprint's World
201 // Some Blueprints have no connected World / GameObjects,
202 // so we need to check if the World is valid
203
204 UWorld* BlueprintDebugWorld = InBlueprint->GetWorldBeingDebugged();
205 if (BlueprintDebugWorld != nullptr)
206 {
207 RegisterUWorldAsset(BlueprintDebugWorld);
208 }
209}
