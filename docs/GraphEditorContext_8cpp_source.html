51{
52 Super::Tick(DeltaTime);
53
54 if (TreeViewCanTick())
55 {
56 TickTreeView();
57
58 TSharedPtr<STreeView<TSharedPtr<FGraphActionNode>>> TreeViewPtr = TreeView.Pin();
59
60 TreeViewTickRequirements.PrevSearchText = FilterTextBox.Pin()->GetText().ToString();
61 TreeViewTickRequirements.PrevNumGeneratedChildren = TreeViewPtr->GetNumGeneratedChildren();
62 TreeViewTickRequirements.PrevNumItemsBeingObserved = TreeViewPtr->GetNumItemsBeingObserved();
63 TreeViewTickRequirements.PrevScrollDistance = TreeViewPtr->GetScrollDistance().Y;
64 }
65
66 return true;

108void UGraphEditorContext::SelectAction(const int32 &InIndex)
109{
110 if (InIndex < 0)
111 return;
112
113 if (!CheckBoxes.IsEmpty() && InIndex < CheckBoxes.Num())
114 {
115 if (CheckBoxes[InIndex].IsValid())
116 {
117 CheckBoxes[InIndex].Pin()->ToggleCheckedState();
118 return;
119 }
120 }
121
122 TSharedPtr<FGraphActionNode> ChosenTreeViewAction = GetTreeViewAction(InIndex - GetStaticIndexOffset());
123 if (!ChosenTreeViewAction.IsValid())
124 {
125 UE_LOG(LogAccession, Warning, TEXT("SelectGraphAction: Provided TreeView Action is Invalid"))
126 return;
127 }
128
129 auto TreeViewPtr = TreeView.Pin();
130 if (ChosenTreeViewAction->IsActionNode())
131 {
132 TreeViewPtr->Private_ClearSelection();
133 TreeViewPtr->Private_SetItemSelection(ChosenTreeViewAction, true, true);
134 TreeViewPtr->Private_SignalSelectionChanged(ESelectInfo::Type::OnMouseClick);
135 }
136 else
137 {
138 TreeViewPtr->Private_OnItemDoubleClicked(ChosenTreeViewAction);
139 }
140}

237bool UGraphEditorContext::FindStaticComponents(const TSharedRef<SWidget> &SearchRoot)
238{
239 TArray<FSlotBase *> FoundComponentSlots = GetWidgetSlotsByType(
240 SearchRoot,
241 TSet<FString>{
242 TEXT("SCheckBox")});
243
244 if (!FoundComponentSlots.IsEmpty())
245 {
246 // Sort and Index the Static Components.
247 for (int i = 0; i < FoundComponentSlots.Num(); i++)
248 {
249 FSlotBase *FoundComponentSlot = FoundComponentSlots[i];
250
251 TSharedPtr<SWidget> DetachedWidget = FoundComponentSlot->DetachWidget();
252 if (!DetachedWidget.IsValid())
253 continue;
254
255 int32 ComponentIndex = -1;
256 FString ComponentType = DetachedWidget->GetTypeAsString();
257
258 if (ComponentType == "SCheckBox")
259 {
260 ComponentIndex = CheckBoxes.Num();
261 CheckBoxes.Add(StaticCastSharedPtr<SCheckBox>(DetachedWidget));
262 }
263
264 FoundComponentSlot->AttachWidget(
265 SNew(SContentIndexer)
266 .IndexValue(ComponentIndex)
267 .IndexPositionToContent(EIndexerPosition::Left)
268 .ContentToIndex(DetachedWidget));
269 }
270
271 return true;
272 }
273
274 return false;
275}

282bool UGraphEditorContext::TreeViewRequiresTick()
283{
284 if (!TreeView.IsValid() || !GraphMenu.IsValid())
285 return false;
286
287 bool bFilterTextChange = FilterTextBox.IsValid()
288 ? FilterTextBox.Pin()->GetText().ToString() != TreeViewTickRequirements.PrevSearchText
289 : false;
290
291 TSharedPtr<STreeView<TSharedPtr<FGraphActionNode>>> TreeViewPtr = TreeView.Pin();
292
293 return (
294 bFilterTextChange ||
295 TreeViewPtr->GetNumItemsBeingObserved() != TreeViewTickRequirements.PrevNumItemsBeingObserved ||
296 TreeViewPtr->GetNumGeneratedChildren() != TreeViewTickRequirements.PrevNumGeneratedChildren ||
297 TreeViewPtr->GetScrollDistance().Y != TreeViewTickRequirements.PrevScrollDistance);
298}

300void UGraphEditorContext::TickTreeView()
301{
302 if (!TreeViewRequiresTick())
303 return;
304
305 TSharedPtr<STreeView<TSharedPtr<FGraphActionNode>>> TreeViewPtr = TreeView.Pin();
306
307 TArray<TSharedPtr<FGraphActionNode>> Items = TArray<TSharedPtr<FGraphActionNode>>(
308 TreeViewPtr->GetRootItems());
309
310 TSharedPtr<STableRow<TSharedPtr<FGraphActionNode>>> ItemWidget = nullptr;
311 const int32 IndexOffset = GetStaticIndexOffset();
312
313 while (Items.Num() > 0)
314 {
315 const TSharedPtr<FGraphActionNode> Item = Items[0];
316 Items.RemoveAt(0);
317
318 if (TreeViewPtr->IsItemExpanded(Item))
319 Items.Append(Item->Children);
320
321 ItemWidget = StaticCastSharedPtr<STableRow<TSharedPtr<FGraphActionNode>>>(
322 TreeViewPtr->WidgetFromItem(Item));
323 if (!ItemWidget.IsValid())
324 continue;
325
326 TSharedPtr<SWidget> ItemContent = ItemWidget->GetContent();
327
328 if (ItemContent->GetType() == "SContentIndexer")
329 {
330 UpdateIndexWidget(
331 StaticCastSharedRef<SContentIndexer>(ItemContent.ToSharedRef()),
332 IndexOffset + ItemWidget->GetIndexInList());
333 }
334 else
335 {
336 ItemWidget->SetContent(
337 CreateIndexWrapper(ItemContent.ToSharedRef(), IndexOffset + ItemWidget->GetIndexInList()));
338 }
339 }
340}
