34void UTranscriptionVisualizer::ConstructVisualizer()
35{
36 TSharedPtr<STranscriptionVis> MenuContent = SNew(STranscriptionVis)
37 .VisAmount(MaxVisualisationAmount);
38
39 MenuContent->ForceVolatile(true);
40
41 FDisplayMetrics DisplayMetrics;
42 FSlateApplication::Get().GetDisplayMetrics(DisplayMetrics);
43
44 FVector2D VisPosition = FVector2D();
45
46 if (FSlateApplication::Get().GetActiveTopLevelRegularWindow().IsValid())
47 {
48 VisPosition = FSlateApplication::Get().GetActiveTopLevelRegularWindow()->GetPositionInScreen();
49 }
50 VisPosition.X = DisplayMetrics.PrimaryDisplayWidth;
51 VisPosition.Y = DisplayMetrics.PrimaryDisplayHeight;
52
53 TSharedRef<SWindow> MenuWindow = SNew(SWindow)
54 .Type(EWindowType::Normal)
55 .SizingRule(ESizingRule::Autosized)
56 .ScreenPosition(VisPosition)
57 .ClientSize(FVector2D(10, 10))
58 .IsPopupWindow(true)
59 //.InitialOpacity(0.5f)
60 .SupportsTransparency(EWindowTransparency::PerWindow)
61 .ActivationPolicy(EWindowActivationPolicy::Always)
62 .AdjustInitialSizeAndPositionForDPIScale(true)
63 [MenuContent.ToSharedRef()];
64
65 TSharedPtr<SWindow> TopLevelWindow = FSlateApplication::Get().GetActiveTopLevelRegularWindow();
66
67 MenuWindow->AssignParentWidget(TopLevelWindow);
68 FSlateApplication::Get().AddWindowAsNativeChild(MenuWindow, TopLevelWindow.ToSharedRef(), true);
69
70 VisWindow = MenuWindow.ToWeakPtr();
71 VisContent = MenuContent.ToWeakPtr();
72}

74void UTranscriptionVisualizer::UpdateVisualizer()
75{
76 if (FSlateApplication::Get().IsActive())
77 {
78 VisWindow.Pin()->ShowWindow();
79
80 MoveVisualizer();
81 }
82 else
83 VisWindow.Pin()->HideWindow();
84
85 if (bIsTranscriptionDirty)
86 {
87 FScopeLock Lock(&TranscriptionQueueMutex);
88
89 for (const auto& TransSlot : TranscriptionList)
90 {
91 VisContent.Pin()->UpdateTopTranscription(TransSlot);
92 }
93
94 bIsTranscriptionDirty = false;
95 }
96
97}

99void UTranscriptionVisualizer::ReparentWindow()
100{
101 TSharedPtr<SWindow> TopLevelActiveWindow = FSlateApplication::Get().GetActiveTopLevelRegularWindow();
102 if (!TopLevelActiveWindow.IsValid())
103 return;
104
105 TSharedPtr<SWindow> VisWindowPtr = VisWindow.Pin();
106
107 if (TopLevelActiveWindow == VisWindow.Pin() ||
108 TopLevelActiveWindow->GetContent() == VisWindowPtr->GetParentWidget())
109 return;
110
111 TSharedPtr<SWindow> PrevParentWindow = VisWindowPtr->GetParentWindow();
112 if (PrevParentWindow.IsValid())
113 {
114 PrevParentWindow->RemoveDescendantWindow(VisWindowPtr.ToSharedRef());
115 }
116
117 VisWindowPtr->AssignParentWidget(TopLevelActiveWindow);
118 TopLevelActiveWindow->AddChildWindow(VisWindowPtr.ToSharedRef());
119}

133void UTranscriptionVisualizer::OnTranscriptionRecieved(TArray<FString> InTranscription)
134{
135 FScopeLock Lock(&TranscriptionQueueMutex);
136
137 FString LastTopItem = FString();
138 for (int i = 0; i < InTranscription.Num(); i++)
139 {
140 if (TranscriptionList.Num() >= MaxVisualisationAmount)
141 {
142 TranscriptionList.RemoveAt(0, 1, false);
143 }
144
145 TranscriptionList.Add(InTranscription[i]);
146 }
147
148 bIsTranscriptionDirty = true;

151bool UTranscriptionVisualizer::GetTopScreenVisualizerPosition(FVector2D &OutPosition)
152{
153 TSharedPtr<SWindow> TopLevelWindow = FSlateApplication::Get().GetActiveTopLevelRegularWindow();
154 if (!TopLevelWindow.IsValid())
155 return false;
156
157 FVector2D ActiveWindowPosition = TopLevelWindow->GetPositionInScreen();
158 FVector2D ActiveWindowBounds = TopLevelWindow->GetClientSizeInScreen();
159
160 TSharedPtr<SWindow> VisWindowPtr = VisWindow.Pin();
161
162 OutPosition.X = (ActiveWindowPosition.X + ActiveWindowBounds.X / 2) - (VisWindowPtr->GetClientSizeInScreen().X / 2);
163 OutPosition.Y = (ActiveWindowPosition.Y + ActiveWindowBounds.Y - 50) - VisWindowPtr->GetClientSizeInScreen().Y;
164
165 return true;
166}
