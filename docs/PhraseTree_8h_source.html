180 void FilterContextStack()
181 {
182 bool bRemoveDerivedContextObjects = false;
183
184 int i = this->ContextObjectStack.Num() - 1;
185 if (i < 0)
186 return;
187
188 UPhraseTreeContextObject *CurrObj = nullptr;
189
190 do
191 {
192 CurrObj = this->ContextObjectStack[i];
193
194 if (CurrObj != nullptr && CurrObj->GetIsActive())
195 {
196 i--;
197 continue;
198 }
199
200 if (CurrObj->IsValidLowLevel())
201 {
202 CurrObj->RemoveFromRoot();
203 CurrObj->MarkAsGarbage();
204 }
205
206 this->ContextObjectStack.RemoveAt(i);
207 i--;
208
209 } while (i > 0);
210
211 CurrObj = nullptr;
212 }
