179 void FilterContextStack()
180 {
181 bool bRemoveDerivedContextObjects = false;
182
183 int i = this->ContextObjectStack.Num() - 1;
184 if (i < 0)
185 return;
186
187 UPhraseTreeContextObject *CurrObj = nullptr;
188
189 do
190 {
191 CurrObj = this->ContextObjectStack[i];
192
193 if (CurrObj != nullptr && CurrObj->GetIsActive())
194 {
195 i--;
196 continue;
197 }
198
199 if (CurrObj->IsValidLowLevel())
200 {
201 CurrObj->RemoveFromRoot();
202 CurrObj->MarkAsGarbage();
203 }
204
205 this->ContextObjectStack.RemoveAt(i);
206 i--;
207
208 } while (i > 0);
209
210 CurrObj = nullptr;
211 }
