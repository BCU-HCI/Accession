80void ULocalizedInputLibrary::KeyboardInputAdd(FParseRecord &Record) {
81 GET_ACTIVE_KEYBOARD_WIDGET(KeyboardFocusedWidget);
82
83 FString WidgetType = KeyboardFocusedWidget->GetTypeAsString();
84
85 UParseStringInput *PhraseInput = Record.GetPhraseInput<UParseStringInput>(TEXT("PHRASE_TO_ADD"));
86 if (PhraseInput == nullptr)
87 return;
88
89 if (WidgetType == "SEditableText")
90 {
91 TSharedPtr<SEditableText> EditableText = StaticCastSharedPtr<SEditableText>(KeyboardFocusedWidget);
92 if (!EditableText.IsValid()) {
93 UE_LOG(LogOpenAccessibilityPhraseEvent, Warning, TEXT("KeyboardInputAdd: CURRENT ACTIVE WIDGET IS NOT OF TYPE - SEditableText"));
94 return;
95 }
96
97 FString CurrText = EditableText->GetText().ToString();
98 EditableText->SetText(
99 FText::FromString(CurrText.TrimStartAndEnd() + TEXT(" ") + PhraseInput->GetValue())
100 );
101 }
102 else if (WidgetType == "SMultiLineEditableText")
103 {
104 TSharedPtr<SMultiLineEditableText> MultiLineEditableText = StaticCastSharedPtr<SMultiLineEditableText>(KeyboardFocusedWidget);
105 if (!MultiLineEditableText.IsValid()) {
106 UE_LOG(LogOpenAccessibilityPhraseEvent, Warning, TEXT("KeyboardInputAdd: CURRENT ACTIVE WIDGET IS NOT OF TYPE - SMultiLineEditableText"));
107 return;
108 }
109
110 FString CurrText = MultiLineEditableText->GetText().ToString();
111 MultiLineEditableText->SetText(
112 FText::FromString(CurrText.TrimStartAndEnd() + TEXT(" ") + PhraseInput->GetValue())
113 );
114 }
115 else UE_LOG(LogOpenAccessibilityPhraseEvent, Warning, TEXT("KeyboardInputAdd: CURRENT ACTIVE WIDGET IS NOT AN INTERFACEABLE TYPE"));
116}
