33TSharedPtr<class SGraphNode> FAccessibilityNodeFactory::CreateNode(UEdGraphNode* InNode) const
34{
35 UE_LOG(LogOpenAccessibility, Display, TEXT("Accessibility Node Factory Used to construct %s"), *InNode->GetName());
36
37 check(InNode);
38
39 // Hack to get around the possible infinite loop of using
40 // this factory to create the node from the factory itself.
41 FEdGraphUtilities::UnregisterVisualNodeFactory(FOpenAccessibilityModule::Get().AccessibilityNodeFactory);
42 TSharedPtr<SGraphNode> OutNode = FNodeFactory::CreateNodeWidget(InNode);
43 FEdGraphUtilities::RegisterVisualNodeFactory(FOpenAccessibilityModule::Get().AccessibilityNodeFactory);
44
45 // Get Node Accessibility Index, from registry
46 TSharedRef<FGraphIndexer> GraphIndexer = FOpenAccessibilityModule::Get()
47 .AssetAccessibilityRegistry->GetGraphIndexer(InNode->GetGraph());
48
49 int NodeIndex = -1;
50 GraphIndexer->GetOrAddNode(InNode, NodeIndex);
51
52 {
53 // Create Accessibility Widgets For Pins
54 TArray<UEdGraphPin*> Pins = InNode->GetAllPins();
55 TSharedPtr<SGraphPin> PinWidget;
56
57 for (int i = 0; i < Pins.Num(); i++)
58 {
59 UEdGraphPin* Pin = Pins[i];
60
61 PinWidget = OutNode->FindWidgetForPin(Pin);
62 if (!PinWidget.IsValid())
63 {
64 continue;
65 }
66
67 WrapPinWidget(Pin, PinWidget.ToSharedRef(), i, OutNode.Get());
68 }
69
70 PinWidget.Reset();
71 }
72
73 // Wrap The Node Widget
74 WrapNodeWidget(InNode, OutNode.ToSharedRef(), NodeIndex);
75
76 return OutNode;
77}

132void FAccessibilityNodeFactory::WrapPinWidget(UEdGraphPin* Pin, TSharedRef<SGraphPin> PinWidget, int PinIndex, SGraphNode* OwnerNode) const
133{
134 TSharedRef<SWidget> PinWidgetContent = PinWidget->GetContent();
135 check(PinWidgetContent != SNullWidget::NullWidget);
136
137 TSharedRef<SWidget> AccessibilityWidget = SNew(SOverlay)
138 .Visibility_Lambda([OwnerNode]() -> EVisibility {
139
140 if (OwnerNode->HasAnyUserFocusOrFocusedDescendants() || OwnerNode->IsHovered() || OwnerNode->GetOwnerPanel()->SelectionManager.IsNodeSelected(OwnerNode->GetNodeObj()))
141 return EVisibility::Visible;
142
143 return EVisibility::Hidden;
144 })
145 + SOverlay::Slot()
146 [
147 SNew(SIndexer)
148 .IndexValue(PinIndex)
149 .TextColor(FLinearColor::White)
150 .BorderColor(FLinearColor::Gray)
151 ];
152
153 switch (Pin->Direction)
154 {
155 case EEdGraphPinDirection::EGPD_Input:
156 {
157 PinWidget->SetContent(
158 SNew(SHorizontalBox)
159 + SHorizontalBox::Slot()
160 .AutoWidth()
161 [
162 PinWidgetContent
163 ]
164 + SHorizontalBox::Slot()
165 .AutoWidth()
166 [
167 AccessibilityWidget
168 ]
169 );
170
171 break;
172 }
173
174 case EEdGraphPinDirection::EGPD_Output:
175 {
176 PinWidget->SetContent(
177 SNew(SHorizontalBox)
178 + SHorizontalBox::Slot()
179 .AutoWidth()
180 [
181 AccessibilityWidget
182 ]
183 + SHorizontalBox::Slot()
184 .AutoWidth()
185 [
186 PinWidgetContent
187 ]
188 );
189 break;
190 }
191
192 default:
193 {
194 UE_LOG(LogOpenAccessibility, Error, TEXT("Pin Direction Not Recognized"));
195 break;
196 }
197 }
