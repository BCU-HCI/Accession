84 def Tick(self, delta_time: float):
85 """Tick Callback for Unreal Engine Slate Post Tick.

88

90 delta_time (float): Time since last tick

100 def HandleTranscriptionRequest(
101 self, recv_message: np.ndarray, metadata: dict = None
102 ):
103 """Handles Incoming Transcription Requests
104
105 Takes the Incoming AudioBuffer, Resamples it to 16kHz and Transcribes it using Whisper.
106
107 Args:
108 recv_message (np.ndarray): ndarray of the incoming audio buffer.
109 metadata (dict, optional): metadata of the incoming audio buffer, if any is recieved. Defaults to None.
110 """
111
112 Log(
113 f"Handling Transcription Request | Message: {recv_message} | Size: {recv_message.size} | Shape: {recv_message.shape}"
114 )
115
116 sample_rate = metadata.get("sample_rate", 48000)
117 num_channels = metadata.get("num_channels", 2)
118
119 message_ndarray = self.audio_resampler.resample(
120 recv_message, sample_rate, num_channels
121 )
122
123 trans_segments, trans_metadata = self.whisper_interface.process_audio_buffer(
124 message_ndarray
125 )
126
127 encoded_segments = [
128 transcription.text.encode() for transcription in trans_segments
129 ]
130
131 Log(f"Encoded Segments: {encoded_segments}")
132
133 if len(encoded_segments) > 0:
134 try:
135 self.com_server.SendMultipartWithMeta(
136 message=encoded_segments, meta=trans_metadata
137 )
138
139 except:
140 Log("Error Sending Encoded Transcription Segments", LogLevel.ERROR)
141
142 else:
143 Log("No Transcription Segments Returned", LogLevel.WARNING)
