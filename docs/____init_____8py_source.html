113 """Handles Incoming Transcription Requests
114
115 Takes the Incoming AudioBuffer, Resamples it to 16kHz and Transcribes it using Whisper.
116
117 Args:
118 recv_message (np.ndarray): ndarray of the incoming audio buffer.
119 metadata (dict, optional): metadata of the incoming audio buffer, if any is recieved. Defaults to None.
120 """
121
122 Log(
123 f"Handling Transcription Request | Message: {recv_message} | Size: {recv_message.size} | Shape: {recv_message.shape}"
124 )
125
126 sample_rate = metadata.get("sample_rate", 48000)
127 num_channels = metadata.get("num_channels", 2)
128
129 message_ndarray = self.audio_resampler.resample(
130 recv_message, sample_rate, num_channels
131 )
132
133 trans_segments, trans_metadata = self.whisper_interface.process_audio_buffer(
134 message_ndarray
135 )
136
137 encoded_segments = [
138 transcription.text.encode() for transcription in trans_segments
139 ]
140
141 Log(f"Encoded Segments: {encoded_segments}")
142
143 if len(encoded_segments) > 0:
144 try:
145 self.com_server.SendMultipartWithMeta(
146 message=encoded_segments, meta=trans_metadata
147 )
148
149 except:
150 Log("Error Sending Encoded Transcription Segments", LogLevel.ERROR)
151
152 else:
153 Log("No Transcription Segments Returned", LogLevel.WARNING)
