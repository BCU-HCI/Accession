257 [[nodiscard]] TArray<FTabManager::FTab> CollectManagedTabs(const TSharedRef<FTabManager>& TabManager)
258 {
259 TArray<FTabManager::FTab> ManagedTabs = TArray<FTabManager::FTab>();
260
261 const TSharedRef<FTabManager::FLayout> Layout = TabManager->PersistLayout();
262 const TSharedPtr<FOpenArea> MainArea = StaticCastSharedPtr<FOpenArea>(Layout->GetPrimaryArea().Pin());
263 if (!MainArea.IsValid())
264 {
265 UE_LOG(LogOpenAccessibility, Warning, TEXT("CollectManagedTabs: Primary Area is Not Valid"))
266 return ManagedTabs;
267 }
268
269 TSharedPtr<FTabManager::FLayoutNode> CurrentNode = nullptr;
270 TArray<TSharedRef<FTabManager::FLayoutNode>> NodesToCheck = TArray{
271 MainArea->GetChildNodes()
272 };
273
274 while(!NodesToCheck.IsEmpty())
275 {
276 CurrentNode = NodesToCheck[0];
277 NodesToCheck.RemoveAt(0);
278
279 if (TSharedPtr<FTabManager::FStack> AsStack = CurrentNode->AsStack())
280 {
281 ManagedTabs.Append(StaticCastSharedPtr<FOpenStack>(AsStack)->GetTabs());
282 }
283 else if (TSharedPtr<FTabManager::FArea> AsArea = CurrentNode->AsArea())
284 {
285 NodesToCheck.Append(StaticCastSharedPtr<FOpenArea>(AsArea)->GetChildNodes());
286 }
287 else if (TSharedPtr<FTabManager::FSplitter> AsSplitter = CurrentNode->AsSplitter())
288 {
289 NodesToCheck.Append(StaticCastSharedPtr<FOpenArea>(AsSplitter)->GetChildNodes());
290 }
291 else
292 {
293 UE_LOG(LogOpenAccessibilityPhraseEvent, Warning, TEXT("CollectManagedTabs: Unknown Node Type."))
294 }
295 }
296
297 return ManagedTabs;
298 }

461void UWindowInteractionLibrary::SelectTabInStack(FParseRecord& Record)
462{
463 UParseStringInput* TabNameInput = Record.GetPhraseInput<UParseStringInput>(TEXT("TAB_NAME"));
464 if (TabNameInput == nullptr)
465 {
466 UE_LOG(LogOpenAccessibilityPhraseEvent, Warning, TEXT("SelectTabInStack: No Tab Name Input Found."))
467 return;
468 }
469
470 GET_ACTIVE_TAB(ActiveTab);
471
472 TSharedPtr<FTabManager> ActiveTabManager = ActiveTab->GetTabManagerPtr();
473 if (!ActiveTabManager.IsValid())
474 {
475 UE_LOG(LogOpenAccessibilityPhraseEvent, Warning, TEXT("SelectTabInStack: Cannot Find Active Tab Manager"))
476 return;
477 }
478
479 TArray<FTabManager::FTab> FoundTabs = TabUtils::CollectManagedTabs(ActiveTabManager.ToSharedRef());
480 if (FoundTabs.IsEmpty())
481 {
482 UE_LOG(LogOpenAccessibilityPhraseEvent, Warning, TEXT("SelectTabInStack: No Tabs Found."))
483 return;
484 }
485
486 FString TargetTabName = TabNameInput->GetValue();
487
488 TSharedPtr<SDockTab> FoundTabWidget = TSharedPtr<SDockTab>();
489 int32 FoundTabDistance = INT32_MAX;
490
491 for (auto& Tab : FoundTabs)
492 {
493 TSharedPtr<SDockTab> CurrTabWidget= ActiveTabManager->FindExistingLiveTab(
494 Tab.TabId
495 );
496
497 if (!CurrTabWidget.IsValid())
498 {
499 continue;
500 }
501
502 FString CurrName = CurrTabWidget->GetTabLabel().ToString().ToUpper();
503 if (CurrName == TargetTabName || CurrName.Contains(TargetTabName))
504 {
505 FoundTabWidget = CurrTabWidget;
506 break;
507 }
508
509 int32 Distance = Algo::LevenshteinDistance(CurrName, TargetTabName);
510 if (Distance < FoundTabDistance)
511 {
512 FoundTabWidget = CurrTabWidget;
513 FoundTabDistance = Distance;
514 }
515 }
516
517 if (!FoundTabWidget.IsValid())
518 {
519 UE_LOG(LogOpenAccessibilityPhraseEvent, Warning, TEXT("SelectTabInStack: No Tab Found."))
520 return;
521 }
522
523 FGlobalTabmanager::Get()->SetActiveTab(FoundTabWidget);
524 FoundTabWidget->ActivateInParent(SetDirectly);
525}
