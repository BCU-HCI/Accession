67 virtual FParseResult ParsePhrase(TArray<FString>& InPhraseArray, FParseRecord& InParseRecord) override
68 {
69 if (InPhraseArray.Num() == 0)
70 {
71 UE_LOG(LogOpenAccessibilityCom, Log, TEXT("|| Emptied Phrase Array ||"))
72
73 return FParseResult(PHRASE_REQUIRES_MORE, AsShared());
74 }
75
76 if (MeetsInputRequirements(InPhraseArray.Last()))
77 {
78 // Get the Input String.
79 FString InputToRecord = InPhraseArray.Pop();
80
81 // Append the Input String to the Record.
82 InParseRecord.AddPhraseString(InputToRecord);
83
84 if (!InputToRecord.IsNumeric() && NumericParser::IsValidNumeric(InputToRecord, false))
85 {
86 NumericParser::StringToNumeric(InputToRecord, false);
87 }
88
89 if (!RecordInput(InputToRecord, InParseRecord))
90 {
91 UE_LOG(LogOpenAccessibilityCom, Log, TEXT("|| Unable to Record Input ||"))
92
93 return FParseResult(PHRASE_UNABLE_TO_PARSE, AsShared());
94 }
95
96 OnPhraseParsed.ExecuteIfBound(InParseRecord);
97
98 return ParseChildren(InPhraseArray, InParseRecord);
99 }
100
101 return FParseResult(PHRASE_UNABLE_TO_PARSE, AsShared());
102 }
