104bool UGraphAddNodeContextMenu::Tick(float DeltaTime)
105{
106 if (!GraphMenu.IsValid() || !Menu.IsValid())
107 return false;
108
109 if (DoesItemsRequireRefresh())
110 RefreshIndexWidgets();
111
112 TSharedPtr<STreeView<TSharedPtr<FGraphActionNode>>> TreeViewPtr = TreeView.Pin();
113
114 // Set Previous Vars For Next Tick
115 PrevFilterString = FilterTextBox.Pin()->GetText().ToString();
116 PrevNumItemsBeingObserved = TreeViewPtr->GetNumItemsBeingObserved();
117 PrevNumGeneratedChildren = TreeViewPtr->GetNumGeneratedChildren();
118 PrevScrollDistance = TreeViewPtr->GetScrollDistance().Y;
119
120 return true;
121}

149bool UGraphAddNodeContextMenu::DoesItemsRequireRefresh()
150{
151 TSharedPtr<STreeView<TSharedPtr<FGraphActionNode>>> TreeViewPtr = TreeView.Pin();
152
153 return (
154 FilterTextBox.Pin()->GetText().ToString() != PrevFilterString ||
155 TreeViewPtr->GetNumItemsBeingObserved() != PrevNumItemsBeingObserved ||
156 TreeViewPtr->GetNumGeneratedChildren() != PrevNumGeneratedChildren ||
157 TreeViewPtr->GetScrollDistance().Y != PrevScrollDistance);

160void UGraphAddNodeContextMenu::RefreshIndexWidgets()
161{
162
163 TSharedPtr<STreeView<TSharedPtr<FGraphActionNode>>> TreeViewPtr = TreeView.Pin();
164
165 TArray<TSharedPtr<FGraphActionNode>> Items = TArray<TSharedPtr<FGraphActionNode>>(TreeViewPtr->GetRootItems());
166
167 {
168 TSharedPtr<STableRow<TSharedPtr<FGraphActionNode>>> ItemWidget = nullptr;
169
170 while (Items.Num() > 0)
171 {
172 const TSharedPtr<FGraphActionNode> Item = Items[0];
173 Items.RemoveAt(0);
174
175 if (TreeViewPtr->IsItemExpanded(Item))
176 Items.Append(Item->Children);
177
178 ItemWidget = StaticCastSharedPtr<STableRow<TSharedPtr<FGraphActionNode>>>(
179 TreeViewPtr->WidgetFromItem(Item));
180
181 if (!ItemWidget.IsValid())
182 continue;
183
184 // TO-DO: Change To Non-HardCoded Type Comparison.
185 if (ItemWidget->GetContent()->GetType() == "SContentIndexer")
186 {
187 UpdateIndexWidget(ItemWidget.ToSharedRef());
188 }
189 else
190 {
191 ApplyIndexWidget(ItemWidget.ToSharedRef());
192 }
193 }
194 }
195}

243void UGraphAddNodeContextMenu::PerformGraphAction(const int32 InIndex)
244{
245 TSharedPtr<FGraphActionNode> GraphAction = GetGraphActionFromIndexSP(InIndex);
246
247 if (!GraphAction.IsValid())
248 {
249 UE_LOG(LogAccession, Warning, TEXT("PerformGraphAction: Provided GraphAction is Invalid."));
250 }
251
252 if (GraphAction->IsActionNode())
253 {
254 TreeView.Pin()->Private_ClearSelection();
255 TreeView.Pin()->Private_SetItemSelection(GraphAction, true, true);
256 TreeView.Pin()->Private_SignalSelectionChanged(ESelectInfo::OnMouseClick);
257 }
258 else
259 {
260 TreeView.Pin()->Private_OnItemDoubleClicked(GraphAction);
261 }
262}
