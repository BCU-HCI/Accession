103bool UGraphAddNodeContextMenu::Tick(float DeltaTime)
104{
105 if (!GraphMenu.IsValid() || !Menu.IsValid())
106 return false;
107
108 if (DoesItemsRequireRefresh())
109 RefreshIndexWidgets();
110
111 TSharedPtr<STreeView<TSharedPtr<FGraphActionNode>>> TreeViewPtr = TreeView.Pin();
112
113 // Set Previous Vars For Next Tick
114 PrevFilterString = FilterTextBox.Pin()->GetText().ToString();
115 PrevNumItemsBeingObserved = TreeViewPtr->GetNumItemsBeingObserved();
116 PrevNumGeneratedChildren = TreeViewPtr->GetNumGeneratedChildren();
117 PrevScrollDistance = TreeViewPtr->GetScrollDistance().Y;
118
119 return true;

148bool UGraphAddNodeContextMenu::DoesItemsRequireRefresh()
149{
150 TSharedPtr<STreeView<TSharedPtr<FGraphActionNode>>> TreeViewPtr = TreeView.Pin();

152 return (
153 FilterTextBox.Pin()->GetText().ToString() != PrevFilterString ||
154 TreeViewPtr->GetNumItemsBeingObserved() != PrevNumItemsBeingObserved ||
155 TreeViewPtr->GetNumGeneratedChildren() != PrevNumGeneratedChildren ||
156 TreeViewPtr->GetScrollDistance().Y != PrevScrollDistance);

159void UGraphAddNodeContextMenu::RefreshIndexWidgets()
160{
161
162 TSharedPtr<STreeView<TSharedPtr<FGraphActionNode>>> TreeViewPtr = TreeView.Pin();
163
164 TArray<TSharedPtr<FGraphActionNode>> Items = TArray<TSharedPtr<FGraphActionNode>>(TreeViewPtr->GetRootItems());
165
166 {
167 TSharedPtr<STableRow<TSharedPtr<FGraphActionNode>>> ItemWidget = nullptr;
168
169 while (Items.Num() > 0)
170 {
171 const TSharedPtr<FGraphActionNode> Item = Items[0];
172 Items.RemoveAt(0);
173
174 if (TreeViewPtr->IsItemExpanded(Item))
175 Items.Append(Item->Children);
176
177 ItemWidget = StaticCastSharedPtr<STableRow<TSharedPtr<FGraphActionNode>>>(
178 TreeViewPtr->WidgetFromItem(Item));
179
180 if (!ItemWidget.IsValid())
181 continue;
182
183 // TO-DO: Change To Non-HardCoded Type Comparison.
184 if (ItemWidget->GetContent()->GetType() == "SContentIndexer")
185 {
186 UpdateIndexWidget(ItemWidget.ToSharedRef());
187 }
188 else
189 {
190 ApplyIndexWidget(ItemWidget.ToSharedRef());
191 }
192 }
193 }

242void UGraphAddNodeContextMenu::PerformGraphAction(const int32 InIndex)
243{
244 TSharedPtr<FGraphActionNode> GraphAction = GetGraphActionFromIndexSP(InIndex);
245
246 if (!GraphAction.IsValid())
247 {
248 UE_LOG(LogAccession, Warning, TEXT("PerformGraphAction: Provided GraphAction is Invalid."));
249 }
250
251 if (GraphAction->IsActionNode())
252 {
253 TreeView.Pin()->Private_ClearSelection();
254 TreeView.Pin()->Private_SetItemSelection(GraphAction, true, true);
255 TreeView.Pin()->Private_SignalSelectionChanged(ESelectInfo::OnMouseClick);
256 }
257 else
258 {
259 TreeView.Pin()->Private_OnItemDoubleClicked(GraphAction);
260 }
