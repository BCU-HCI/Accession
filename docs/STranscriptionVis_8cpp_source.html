13void STranscriptionVis::Construct(const FArguments &InArgs)
14{
15 // Transcription Holder
16 TSharedPtr<SVerticalBox> TranscriptionHolder = SNew(SVerticalBox) + SVerticalBox::Slot()
17 .Padding(4.0f)
18 .AutoHeight();
19
20 // Verify a least one slot will be constructed
21 int TranscriptionSlotAmount = FMath::Max(1, InArgs._VisAmount);
22
23 FSlateFontInfo FontInfo = FAppStyle::GetFontStyle("NormalText");
24 FontInfo.Size = 12;
25
26 TSharedPtr<STextBlock> CurrentTranscriptionSlot;
27 for (int i = 0; i < TranscriptionSlotAmount; i++)
28 {
29 TranscriptionHolder->AddSlot()
30 .HAlign(HAlign_Center)
31 .Padding(4.0f)
32 .AutoHeight()
33 [SAssignNew(CurrentTranscriptionSlot, STextBlock)
34 .Text(FText::GetEmpty())
35 .Font(FontInfo)
36 .SimpleTextMode(true)
37 .ColorAndOpacity(i == 0 ? FSlateColor(FLinearColor(1.0f, 1.0f, 0, 1.0f)) : FSlateColor(FLinearColor(0.5f, 0.5f, 0.5f, 1.0f)))];
38
39 TranscriptionSlots.Add(CurrentTranscriptionSlot);
40 }
41
42 // Construct the Main Component
43
44 UAudioManager *AudioManager = FAccessionCommunicationModule::Get().AudioManager;
45
46 ChildSlot
47 .Padding(FMargin(5.0f))
48 [SNew(SOverlay) + SOverlay::Slot()
49 .ZOrder(1)
50 [SNew(SBorder)
51 .BorderBackgroundColor(FSlateColor(FLinearColor::Gray))
52 .BorderBackgroundColor_Lambda([AudioManager]()
53 { return AudioManager != nullptr && AudioManager->IsCapturingAudio()
54 ? FSlateColor(FLinearColor::Red)
55 : FSlateColor(FLinearColor::Gray); })
56 [SNew(SBox)
57 .MinDesiredWidth(250.0f)
58 .MinDesiredHeight(60.0f)
59 [TranscriptionHolder.ToSharedRef()]]]];
60
61 this->TranscriptionContainer = TranscriptionHolder;
62}

69void STranscriptionVis::UpdateTopTranscription(const FString &InTopTranscription)
70{
71 FString LastTopText = InTopTranscription;
72 FString TempText;
73
74 TSharedPtr<STextBlock> CurrentTranscriptionSlot;
75 for (TWeakPtr<STextBlock> &TranscriptionSlot : TranscriptionSlots)
76 {
77 CurrentTranscriptionSlot = TranscriptionSlot.Pin();
78
79 TempText = FString(CurrentTranscriptionSlot->GetText().ToString());
80 CurrentTranscriptionSlot->SetText(FText::FromString(LastTopText));
81
82 CurrentTranscriptionSlot->Invalidate(EInvalidateWidgetReason::PaintAndVolatility);
83
84 LastTopText = TempText;
85 }
86
87 TranscriptionContainer.Pin()->Invalidate(EInvalidateWidget::Layout);
88}
