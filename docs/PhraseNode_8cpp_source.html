64FParseResult FPhraseNode::ParsePhrase(TArray<FString> &InPhraseArray,
65 FParseRecord &InParseRecord)
66{
67 if (InPhraseArray.IsEmpty())
68 {
69 UE_LOG(LogAccessionCom, Log, TEXT("|| Emptied Phrase Array ||"))
70
71 return FParseResult(PHRASE_REQUIRES_MORE, AsShared());
72 }
73
74 // Pop the Phrase Linked to this Node.
75 // Apply to the Record.
76 FString LinkedPhrase = InPhraseArray.Pop();
77
78 // Append Removed Phrase To Record.
79 InParseRecord.AddPhraseString(LinkedPhrase);
80
81 OnPhraseParsed.ExecuteIfBound(InParseRecord);
82
83 // Pass
84 return ParseChildren(InPhraseArray, InParseRecord);
85}

124bool FPhraseNode::BindChildNode(TPhraseNode InNode)
125{
126 if (!InNode.IsValid())
127 return false;

129 for (auto &ChildNode : ChildNodes)

131 if (ChildNode->RequiresPhrase(InNode->BoundPhrase))
132 {
133 return ChildNode->BindChildrenNodes(InNode->ChildNodes);

188FParseResult FPhraseNode::ParseChildren(TArray<FString> &InPhraseArray, FParseRecord &InParseRecord)
189{
190 if (HasLeafChild())
191 return ChildNodes[0]->ParsePhrase(InPhraseArray, InParseRecord);
192 if (InPhraseArray.IsEmpty())
193 return FParseResult(PHRASE_REQUIRES_MORE, AsShared());
194
195 // Below Can Be Optimized.
196 // Maybe bypass the loop if Distance == 0 and Sort ChildNodes with Derrived PhraseNodes Last?
197
198 int FoundChildIndex = -1;
199 {
200 int32 FoundChildDistance = INT32_MAX, CurrentDistance = INT32_MAX;
201
202 for (int i = 0; i < ChildNodes.Num(); i++)
203 {
204 // Child Nodes Require Unique Phrases to Siblings.
205 if (ChildNodes[i]->RequiresPhrase(InPhraseArray.Last(), CurrentDistance))
206 {
207 if (FoundChildDistance > CurrentDistance)
208 {
209 FoundChildIndex = i;
210 FoundChildDistance = CurrentDistance;
211 }
212 }
213 }
214 }
215
216 if (FoundChildIndex != -1)
217 {
218 return ChildNodes[FoundChildIndex]->ParsePhrase(InPhraseArray, InParseRecord);
219 }
220
221 /*else if (!InPhraseArray.IsEmpty())
222 {
223 return FParseResult(PHRASE_REQUIRES_MORE_CORRECT_PHRASES, AsShared());
224 }*/
225
226 return FParseResult(PHRASE_UNABLE_TO_PARSE, AsShared());
