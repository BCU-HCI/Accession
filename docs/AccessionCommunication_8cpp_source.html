24void FAccessionCommunicationModule::StartupModule()
25{
26 LoadZMQDLL();
27
28 // This code will execute after your module is loaded into memory; the exact timing is specified in the .uplugin file per-module
29 UE_LOG(LogAccessionCom, Display, TEXT("AccessionComModule::StartupModule()"));
30
31 // Initialize AudioManager
32 AudioManager = NewObject<UAudioManager>();
33 AudioManager->AddToRoot();
34
35 AudioManager->OnAudioReadyForTranscription
36 .BindRaw(this, &FAccessionCommunicationModule::TranscribeWaveForm);
37
38 // Initialize Socket Server
39 SocketServer = MakeShared<FSocketCommunicationServer>();
40
41 // Build The Phrase Tree
42 BuildPhraseTree();
43
44 // Bind Tick Event
45 TickDelegate = FTickerDelegate::CreateRaw(this, &FAccessionCommunicationModule::Tick);
46 TickDelegateHandle = FTSTicker::GetCoreTicker().AddTicker(TickDelegate);
47
48 // Bind Input Events
49 KeyDownEventHandle = FSlateApplication::Get().OnApplicationPreInputKeyDownListener().AddRaw(this, &FAccessionCommunicationModule::HandleKeyDownEvent);
50
51 // Register Console Commands
52 RegisterConsoleCommands();
53}

71bool FAccessionCommunicationModule::Tick(const float DeltaTime)
72{
73 // Detect if any events are ready to be received.
74 if (SocketServer->EventOccured())
75 {
76 TArray<FString> RecvStrings;
77 TSharedPtr<FJsonObject> RecvMetadata;
78
79 // Receive the Detected Event, with separate transcriptions and metadata.
80 if (SocketServer->RecvStringMultipartWithMeta(RecvStrings, RecvMetadata))
81 {
82 OA_LOG(LogAccessionCom, Log, TEXT("TRANSCRIPTION RECEIVED"), TEXT("Received Multipart - Message Count: %d"), RecvStrings.Num());
83 for (int i = 0; i < RecvStrings.Num(); i++)
84 {
85 OA_LOG(LogAccessionCom, Log, TEXT("TRANSCRIPTION RECEIVED"), TEXT("Transcription - %d: { %s }"), i, *RecvStrings[i]);
86 }
87
88 // Send Received Transcriptions to any bound events.
89 OnTranscriptionRecieved.Broadcast(RecvStrings);
90 }
91 }
92
93 return true;
94}

96void FAccessionCommunicationModule::HandleKeyDownEvent(const FKeyEvent &InKeyEvent)
97{
98 FKey EventKey = InKeyEvent.GetKey();
99
100 bool isTargetKey = EventKey == EKeys::LeftAlt || EventKey == EKeys::RightAlt;
101
102 // If the Voice Command Key is Pressed, Toggle Audio Capture.
103 if (isTargetKey && !InKeyEvent.IsRepeat())
104 {
105 if (!AudioManager->IsCapturingAudio())
106 {
107 OA_LOG(LogAccessionCom, Log, TEXT("AudioCapture Change"), TEXT("Starting Audio Capture"));
108 AudioManager->StartCapturingAudio();
109 }
110 else
111 {
112 OA_LOG(LogAccessionCom, Log, TEXT("AudioCapture Change"), TEXT("Stopping Audio Capture"));
113 AudioManager->StopCapturingAudio();
114 }
115 }
116}

118void FAccessionCommunicationModule::TranscribeWaveForm(const TArray<float> AudioBufferToTranscribe)
119{
120 if (AudioBufferToTranscribe.Num() == 0)
121 {
122 UE_LOG(LogAccessionCom, Warning, TEXT("|| Transcription Ready || Audio Buffer is Empty ||"));
123 return;
124 }
125
126 PrevAudioBuffer = TArray(AudioBufferToTranscribe);
127
128 UE_LOG(LogAccessionCom, Log, TEXT("|| WaveForm Transcription || Array Size: %d || Byte Size: %s ||"), AudioBufferToTranscribe.Num(), *FString::FromInt(AudioBufferToTranscribe.Num() * sizeof(float)));
129
130 // Create Metadata of Audio Source.
131 TSharedPtr<FJsonObject> AudioBufferMetadata = MakeShared<FJsonObject>();
132 AudioBufferMetadata->SetNumberField(TEXT("sample_rate"), AudioManager->GetAudioCaptureSampleRate());
133 AudioBufferMetadata->SetNumberField(TEXT("num_channels"), AudioManager->GetAudioCaptureNumChannels());
134
135 bool bArrayMessageSent = SocketServer->SendArrayMessageWithMeta(AudioBufferToTranscribe, AudioBufferMetadata.ToSharedRef(), ComSendFlags::none);
136
137 OA_LOG(LogAccessionCom, Log, TEXT("TRANSCRIPTION SENT"), TEXT("{%s} Send Audiobuffer (float x %d / %d Hz / %d channels)"),
138 bArrayMessageSent ? TEXT("Success") : TEXT("Failed"),
139 AudioBufferToTranscribe.Num(), AudioManager->GetAudioCaptureSampleRate(), AudioManager->GetAudioCaptureNumChannels());
140}

194void FAccessionCommunicationModule::LoadZMQDLL()
195{
196 FString BaseDir = IPluginManager::Get().FindPlugin("Accession")->GetBaseDir();
197
198 FString LibraryPath;
199#if PLATFORM_WINDOWS
200#if UE_BUILD_DEBUG
201 LibraryPath = FPaths::Combine(*BaseDir, TEXT("Binaries/ThirdParty/ZeroMQ/Win64/libzmq-mt-gd-4_3_5.dll"));
202#else
203 LibraryPath = FPaths::Combine(*BaseDir, TEXT("Binaries/ThirdParty/ZeroMQ/Win64/libzmq-mt-4_3_5.dll"));
204#endif
205#elif PLATFORM_LINUX
206 LibraryPath = FPaths::Combine(*BaseDir, TEXT("Binaries/ThirdParty/ZeroMQ/Linux/libzmq-mt-4_3_5.so"))
207#elif PLATFORM_MAC
208 LibraryPath = FPaths::Combine(*BaseDir, TEXT("Source/ThirdParty/ZeroMQ/Mac/libzmq-mt-4_3_5.dylib"))
209#endif
210
211 ZMQDllHandle = !LibraryPath.IsEmpty() ? FPlatformProcess::GetDllHandle(*LibraryPath) : nullptr;
212
213 if (ZMQDllHandle)
214 {
215 UE_LOG(LogAccessionCom, Log, TEXT("|| LoadZMQDLL || Successfully Loaded ZMQ DLL ||"));
216 }
217 else
218 {
219 UE_LOG(LogAccessionCom, Error, TEXT("|| LoadZMQDLL || Failed to Load ZMQ DLL ||"));
220 }
221}
