46void UAudioManager::StopCapturingAudio()
47{
48 bIsCapturingAudio = false;
49
50 if (AudioBuffer.Num() == 0)
51 return;
52
53 SaveAudioBufferToWAV(Settings.SavePath);
54
55 if (OnAudioReadyForTranscription.ExecuteIfBound(AudioBuffer))
56 {
57 UE_LOG(LogOpenAccessibilityCom, Log, TEXT("|| Executing Audio Ready For Transcription Delegate. ||"));
58 }
59 else
60 {
61 UE_LOG(LogOpenAccessibilityCom, Warning, TEXT("|| No Delegates Bound to Audio Ready For Transcription Delegate. ||"));
62 }
63
64 AudioBuffer.Empty();
65}

77void UAudioManager::SaveAudioBufferToWAV(const FString& FilePath)
78{
79 UE_LOG(LogOpenAccessibilityCom, Log, TEXT("Starting to Save Audio Buffer to WAV"));
80
81 Audio::FSampleBuffer SampleBuffer = Audio::FSampleBuffer(AudioBuffer.GetData(), AudioBuffer.Num(), AudioCapture->GetNumChannels(), AudioCapture->GetSampleRate());
82
83 FileWriter->BeginWriteToWavFile(SampleBuffer, Settings.SaveName, const_cast<FString&>(FilePath), []() {
84 UE_LOG(LogOpenAccessibilityCom, Log, TEXT("Audio Buffer Saved to WAV"));

96void UAudioManager::RegisterAudioGenerator()
97{
98 // Add Audio Generator Delegate to get audio data from stream,
99 // and apply wrapper function due to wanting to reference class function.
100 OnAudioGenerateHandle = AudioCapture->AddGeneratorDelegate(FOnAudioGenerate([this](const float* InAudio, int32 NumSamples) {
101 if (this->IsCapturingAudio()) this->PRIVATE_OnAudioGenerate(InAudio, NumSamples);
102 }));
103}
104
105void UAudioManager::UnregisterAudioGenerator()
106{
107 AudioCapture->RemoveGeneratorDelegate(OnAudioGenerateHandle);
108}
