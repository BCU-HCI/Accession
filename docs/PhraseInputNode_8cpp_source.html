65template<typename InputType>
66FParseResult FPhraseInputNode<InputType>::ParsePhrase(TArray<FString>& InPhraseArray, FParseRecord& InParseRecord)
67{
68 if (InPhraseArray.Num() == 0)
69 {
70 UE_LOG(LogOpenAccessibilityCom, Log, TEXT("|| Emptied Phrase Array ||"))
71
72 return FParseResult(PHRASE_REQUIRES_MORE, AsShared());
73 }
74
75 if (MeetsInputRequirements(InPhraseArray.Last()))
76 {
77 // Get the Input String.
78 FString InputToRecord = InPhraseArray.Pop();
79
80 // Append the Input String to the Record.
81 InParseRecord.AddPhraseString(InputToRecord);
82
83 if (!InputToRecord.IsNumeric() && NumericParser::IsValidNumeric(InputToRecord, false))
84 {
85 NumericParser::StringToNumeric(InputToRecord, false);
86 }
87
88 if (!RecordInput(InputToRecord, InParseRecord))
89 {
90 UE_LOG(LogOpenAccessibilityCom, Log, TEXT("|| Unable to Record Input ||"))
91
92 return FParseResult(PHRASE_UNABLE_TO_PARSE, AsShared());
93 }
94
95 OnPhraseParsed.ExecuteIfBound(InParseRecord);
96
97 return ParseChildren(InPhraseArray, InParseRecord);
98 }
99
100 return FParseResult(PHRASE_UNABLE_TO_PARSE, AsShared());
101}

115bool FPhraseInputNode<int32>::RecordInput(const FString& InInput, FParseRecord& OutParseRecord)
116{
117 int32 Input = FCString::Atoi(*InInput);
118
119 UParseIntInput* ParseInput = MakeParseInput<UParseIntInput>();
120 ParseInput->SetValue(Input);
121
122 OutParseRecord.AddPhraseInput(BoundPhrase, ParseInput);
123
124 OnInputReceived.ExecuteIfBound(Input);
125
126 return true;
