207 [[nodiscard]] FORCEINLINE FString RemoveWordsFromEnd(const FString& InString, const int32& AmountToRemove)
208 {
209 TArray<FString> SplitTextBoxString;
210 InString.ParseIntoArrayWS(SplitTextBoxString);
211
212 int RemovedAmount = 0;
213 int CurrentIndex = SplitTextBoxString.Num() - 1;
214 while (RemovedAmount < AmountToRemove) {
215 if (SplitTextBoxString.IsEmpty())
216 break;
217
218 SplitTextBoxString.RemoveAt(CurrentIndex--);
219 RemovedAmount++;
220 }
221
222 if (SplitTextBoxString.Num() > 0)
223 return FString::Join(SplitTextBoxString, TEXT(" "));
224
225 return TEXT("");
226 }
227}
