88 TArray<TSharedPtr<FGraphQTNode>>& PartitionSpace(FVector2D MinSegmentSize = FVector2D(300, 250))
89 {
90 if (ContainsSegments())
91 return Children;
92
93 FVector2D SegmentSize = (BotRight - TopLeft) / 2;
94 if (SegmentSize.ComponentwiseAllLessThan(MinSegmentSize)) // Ensure New Segments Won't Be Too Small.
95 return Children;
96
97 FVector2D LocalSegmentSize = (LocalBotRight - LocalTopLeft) / 2;
98
99 // Size Array For Incoming Segments
100 Children.Reserve(4);
101
102 for (int Y = 0; Y < 2; ++Y)
103 {
104 for (int X = 0; X < 2; ++X)
105 {
106 FVector2D SegmentOffset(X * SegmentSize.X, Y * SegmentSize.Y);
107 FVector2D LocalSegmentOffset(X * SegmentSize.X, Y * SegmentSize.Y);
108
109 Children.Add(
110 MakeShared<FGraphQTNode>(
111 Owner,
112 TopLeft + SegmentOffset,
113 TopLeft + SegmentOffset + SegmentSize,
114 LocalTopLeft + LocalSegmentOffset,
115 LocalTopLeft + LocalSegmentOffset + LocalSegmentSize,
116 Depth + 1
117 )
118 );
119 }
120 }
121
122 return Children;
123 }

133 void Visualize(FSlateWindowElementList& ElementList, TArray<FVector2D>& LinePoints, const FPaintGeometry& PaintGeometry, const int32& LayerID, const FLinearColor& LineColor = FLinearColor::Green)
134 {

142
143 FSlateDrawElement::MakeLines(ElementList, LayerID, PaintGeometry, LinePoints, ESlateDrawEffect::None, LineColor);
144
145 // Draw Horizontal Line
146 LinePoints[0] = LocalTopLeft + FVector2D(0, (LocalBotRight.Y - LocalTopLeft.Y ) / 2);
147 LinePoints[1] = LocalTopLeft + FVector2D(LocalBotRight.X - LocalTopLeft.X, (LocalBotRight.Y - LocalTopLeft.Y ) / 2);
148
149 FSlateDrawElement::MakeLines(ElementList, LayerID, PaintGeometry, LinePoints, ESlateDrawEffect::None, LineColor);
150
