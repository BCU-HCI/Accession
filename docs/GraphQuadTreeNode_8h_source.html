89 TArray<TSharedPtr<FGraphQTNode>>& PartitionSpace(FVector2D MinSegmentSize = FVector2D(300, 250))
90 {
91 if (ContainsSegments())
92 return Children;
93
94 FVector2D SegmentSize = (BotRight - TopLeft) / 2;
95 if (SegmentSize.ComponentwiseAllLessThan(MinSegmentSize)) // Ensure New Segments Won't Be Too Small.
96 return Children;
97
98 FVector2D LocalSegmentSize = (LocalBotRight - LocalTopLeft) / 2;
99
100 // Size Array For Incoming Segments
101 Children.Reserve(4);
102
103 for (int Y = 0; Y < 2; ++Y)
104 {
105 for (int X = 0; X < 2; ++X)
106 {
107 FVector2D SegmentOffset(X * SegmentSize.X, Y * SegmentSize.Y);
108 FVector2D LocalSegmentOffset(X * SegmentSize.X, Y * SegmentSize.Y);
109
110 Children.Add(
111 MakeShared<FGraphQTNode>(
112 Owner,
113 TopLeft + SegmentOffset,
114 TopLeft + SegmentOffset + SegmentSize,
115 LocalTopLeft + LocalSegmentOffset,
116 LocalTopLeft + LocalSegmentOffset + LocalSegmentSize,
117 Depth + 1
118 )
119 );
120 }
121 }
122
123 return Children;
124 }

134 void Visualize(FSlateWindowElementList& ElementList, TArray<FVector2D>& LinePoints, const FPaintGeometry& PaintGeometry, const int32& LayerID, const FLinearColor& LineColor = FLinearColor::Green)
135 {
136 // Draw Children
137 if (Children.IsEmpty())
138 return;
139
140 // Draw Vertical Line
141 LinePoints[0] = LocalTopLeft + FVector2D((LocalBotRight.X - LocalTopLeft.X) / 2, 0);
142 LinePoints[1] = LocalTopLeft + FVector2D((LocalBotRight.X - LocalTopLeft.X) / 2, LocalBotRight.Y - LocalTopLeft.Y);
143
144 FSlateDrawElement::MakeLines(ElementList, LayerID, PaintGeometry, LinePoints, ESlateDrawEffect::None, LineColor);
145
146 // Draw Horizontal Line
147 LinePoints[0] = LocalTopLeft + FVector2D(0, (LocalBotRight.Y - LocalTopLeft.Y ) / 2);
148 LinePoints[1] = LocalTopLeft + FVector2D(LocalBotRight.X - LocalTopLeft.X, (LocalBotRight.Y - LocalTopLeft.Y ) / 2);
149
150 FSlateDrawElement::MakeLines(ElementList, LayerID, PaintGeometry, LinePoints, ESlateDrawEffect::None, LineColor);
151
152 // Draw Children
153
154 for (auto& Child : Children)
155 {
156 Child->Visualize(ElementList, LinePoints, PaintGeometry, LayerID, LineColor);
157 }

159 }
